190 class Hash
191 {
192 public:
193  static const int HASHSIZE = _details::_HashSize<HASH_METHOD>::size;
194  typedef typename _details::_HashTrait<HASH_METHOD>::type HashType;
195 protected:
196  HashType Hasher;
197 public:
198  INLFUNC void Reset()
199  { try{ Hasher.clear(); return; }
200  catch(std::exception& e){ _LOG_EXPCEPTION(e.what()); }
201  }
202  INLFUNC void Update(LPCVOID data, SIZE_T size)
203  { try{ Hasher.update((LPCBYTE)data, size); return; }
204  catch(std::exception& e){ _LOG_EXPCEPTION(e.what()); }
205  }
206  INLFUNC void Finalize(LPVOID HashValue)
207  { try{ Hasher.final((LPBYTE)HashValue); return; }
208  catch(std::exception& e){ _LOG_EXPCEPTION(e.what()); }
209  }
210  INLFUNC void Calculate(LPCVOID data, SIZE_T size, LPVOID HashValue)
211  { try
212  { Hasher.clear();
213  Hasher.update((LPCBYTE)data, size);
214  Hasher.final((LPBYTE)HashValue);
215  }
216  catch(std::exception& e){ _LOG_EXPCEPTION(e.what()); }
217  }
218  INLFUNC void Calculate(LPCVOID data, SIZE_T size, rt::String& hash_hex)
219  {
220  BYTE hash[HASHSIZE];
221  Calculate(data,size,hash);
222 
223  hash_hex.SetLength(HASHSIZE*2);
224  os::Base16Encode(hash_hex.Begin(), hash, HASHSIZE);
225  }
