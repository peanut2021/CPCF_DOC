{
_LOG(token);
}
_LOG(code.TrimAfter('*'));
_LOG(code.TrimBefore('.'));
_LOG(code.TrimAfterReverse('*'));
_LOG(code.TrimBeforeReverse('.'));
token.Empty();
rt::String_Ref nontoken;
while (code.GetNextToken(symbol, token, nontoken))
{
_LOG(token << '\t' << nontoken);
}
}
