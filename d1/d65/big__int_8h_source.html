58 INLFUNC void BigIntToString(const BigInt& x, rt::String& out)
59 {
60  out = "0x";
61  if(x.is_nonzero())
62  {
63  auto& raw = x.get_word_vector();
64  LPCBYTE end = (LPCBYTE)raw.data();
65  LPCBYTE p = end + raw.size()*sizeof(Botan::word) - 1;
66  for(; p>=end; p--)
67  {
68  BYTE v = *p;
69  int c1 = v>>4;

72  out += (char)((c1>9)?('a'+c2-10):('0'+c2));
