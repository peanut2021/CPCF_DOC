81  template<typename T>
82  INLFUNC static int ftoa(T value, char* str, int prec = 2)
83  { static const T pow10[] = {1, 10, 100, 1000, 10000, 100000, 1000000, 10000000, 100000000, 1000000000};
84  // Hacky test for NaN
85  // under -fast-math this won't work, but then you also won't have correct nan values anyways.
86  // The alternative is to link with libmath (bad) or hack IEEE double bits (bad)
87  if (! (value == value)){ str[0] = 'n'; str[1] = 'a'; str[2] = 'n'; str[3] = '\0'; return 3; }
88  // if input is larger than thres_max, revert to exponential
89  const T thres_max = (T)(0x7FFFFFFF); T diff = 0;
90  char* wstr = str;
91  if (prec < 0){ prec = 0; }
92  else if (prec > 9){ prec = 9; } /* precision of >= 10 can lead to overflow errors */
93  // we'll work in positive values and deal with the negative sign issue later
94  int neg = 0;
95  if(value < 0){ neg = 1; value = -value; }
96  int whole = (int) value; T tmp = (value - whole) * pow10[prec];
97  UINT frac = (UINT)(tmp); diff = tmp - frac;
98  if (diff > 0.5) {
99  ++frac;
100  /* handle rollover, e.g. case 0.99 with prec 1 is 1.0 */
101  if(frac >= pow10[prec]){ frac = 0; ++whole; }
102  } else if (diff == 0.5 && ((frac == 0) || (frac & 1))) {
103  ++frac; /* if halfway, round up if odd, OR if last digit is 0. That last part is strange */
104  }
105  // for very large numbers switch back to native sprintf for exponentials. anyone want to write code to replace this?
106  // normal printf behavior is to print EVERY whole number digit which can be 100s of characters overflowing your buffers == bad
107  if (value > thres_max) {
108  char fmt[5] = "%.2e";
109  fmt[2] = prec + '0';
110  return sprintf(str, fmt, (double)(neg ? -value : value));
111  }
112  if(prec == 0) {
113  diff = value - whole;
114  if (diff > 0.5){ ++whole; } /* greater than 0.5, round up, e.g. 1.6 -> 2 */
115  else if (diff == 0.5 && (whole & 1)){ ++whole; } // exactly 0.5 and ODD, then round up. 1.5 -> 2, but 2.5 -> 2
116  } else {
117  int count = prec; // now do fractional part, as an unsigned number
118  do {
119  --count;
120  *wstr++ = (char)(48 + (frac % 10));
121  } while (frac /= 10);
122  while (count-- > 0) *wstr++ = '0'; // add extra 0s
123  *wstr++ = '.'; // add decimal
124  }
125  do *wstr++ = (char)(48 + (whole % 10)); while (whole /= 10);
126  if(neg){ *wstr++ = '-'; } *wstr='\0';
127  strreverse(str, wstr-1); return (int)(wstr-str);
128  }
