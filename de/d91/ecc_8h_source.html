234 class _PublicKeyOp
235 {
236 protected:
237  rt::ObjectPlaceHolder<typename _details::ECC<CRYTOMETHOD>::PublicKey> _PK;
238 public:
239  bool SetPublicKey(const typename Keypair<CRYTOMETHOD>::PublicKey& pk)
240  { try{
241  Botan::EC_Group group(_details::ECC<CRYTOMETHOD>::ECGroupName());
242  auto& curve = group.get_curve();
243 
244  Botan::BigInt x, y;
245  Botan::BigIntAssign(x, pk);
246 
247  //y = Botan::decompress_point(true, x, curve.get_curve());
248  Botan::BigInt xpow3 = x * x * x;
249  const Botan::BigInt& p = curve.get_p();
250 
251  Botan::BigInt g = curve.get_a() * x;
252  g += xpow3;
253  g += curve.get_b();
254  g = g % p;
255 
256  y = Botan::ressol(g, p);
257 
258  if(y < 0)throw Botan::Illegal_Point("error during EC point decompression");
259  if(y.get_bit(0) != x.get_bit(0))y = p - y;
260 
261  _PK.Reinit(group, Botan::PointGFp(curve, x, y));
262  return true;
263  }catch(...){}
264  return false;
265  }
