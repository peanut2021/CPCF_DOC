382 FORCEINL int PopCount(ULONGLONG x) {
383  x -= (x >> 1) & _details::m1; //put count of each 2 bits into those 2 bits
384  x = (x & _details::m2) + ((x >> 2) & _details::m2); //put count of each 4 bits into those 4 bits
385  x = (x + (x >> 4)) & _details::m4; //put count of each 8 bits into those 8 bits
386  x += x >> 8; //put count of each 16 bits into their lowest 8 bits
387  x += x >> 16; //put count of each 32 bits into their lowest 8 bits
388  x += x >> 32; //put count of each 64 bits into their lowest 8 bits
389  return x & 0x7f;
390 }

403  // This version is for very small inputs (< 16 bytes)
404  FORCEINL unsigned int xxHash_small(const void* key, int len, unsigned int seed)
405  { const unsigned char* p = (unsigned char*)key;
406  const unsigned char* const bEnd = p + len;
407  unsigned int idx = seed + PRIME1;
408  unsigned int crc = PRIME5;
409  const unsigned char* const limit = bEnd - 4;
410  while (p<limit)
411  { crc += ((*(unsigned int*)p) + idx++);
412  crc += _rotl(crc, 17) * PRIME4;
413  crc *= PRIME1;
414  p+=4;
415  }
416  while (p<bEnd)
417  { crc += ((*p) + idx++);
418  crc *= PRIME1;
419  p++;
420  }
421  crc += len;
422  crc ^= crc >> 15;
423  crc *= PRIME2;
424  crc ^= crc >> 13;
425  crc *= PRIME3;
426  crc ^= crc >> 16;
427  return crc;
428  }

431 FORCEINL UINT xxHash(const void* input, int len, unsigned int seed = 0)
432 { // Special case, for small inputs
433  if (len < 16) return _details::xxHash_small(input, len, seed);
434  const unsigned char* p = (const unsigned char*)input;
435  const unsigned char* const bEnd = p + len;
436  unsigned int v1 = seed + _details::PRIME1;
437  unsigned int v2 = v1 * _details::PRIME2 + len;
438  unsigned int v3 = v2 * _details::PRIME3;
439  unsigned int v4 = v3 * _details::PRIME4;
440  const unsigned char* const limit = bEnd - 16;
441  unsigned int crc;
442  while (p<limit)
443  { v1 = _rotl(v1, 13) + (*(unsigned int*)p); p+=4;
444  v2 = _rotl(v2, 11) + (*(unsigned int*)p); p+=4;
445  v3 = _rotl(v3, 17) + (*(unsigned int*)p); p+=4;
446  v4 = _rotl(v4, 19) + (*(unsigned int*)p); p+=4;
447  }
448 
449  p = bEnd - 16;
450  v1 += _rotl(v1, 17); v2 += _rotl(v2, 19); v3 += _rotl(v3, 13); v4 += _rotl(v4, 11);
451  v1 *= _details::PRIME1; v2 *= _details::PRIME1; v3 *= _details::PRIME1; v4 *= _details::PRIME1;
452  v1 += *(unsigned int*)p; p+=4; v2 += *(unsigned int*)p; p+=4; v3 += *(unsigned int*)p; p+=4; v4 += *(unsigned int*)p; // p+=4;
453  v1 *= _details::PRIME2; v2 *= _details::PRIME2; v3 *= _details::PRIME2; v4 *= _details::PRIME2;
454  v1 += _rotl(v1, 11); v2 += _rotl(v2, 17); v3 += _rotl(v3, 19); v4 += _rotl(v4, 13);
455  v1 *= _details::PRIME3; v2 *= _details::PRIME3; v3 *= _details::PRIME3; v4 *= _details::PRIME3;
456 
457  crc = v1 + _rotl(v2, 3) + _rotl(v3, 6) + _rotl(v4, 9);
458  crc ^= crc >> 11;
459  crc += (_details::PRIME4+len) * _details::PRIME1;
460  crc ^= crc >> 15;
461  crc *= _details::PRIME2;
462  crc ^= crc >> 13;
463 
464  return crc;
465 }
