63 template<typename SEQNO>
64 class SeqNoDeJitter
65 {
66 #if defined(PLATFORM_DEBUG_BUILD)
67  rt::hash_set<SEQNO> _Dedup;
68 #endif
69  SEQNO _Min; // consolidate on min
70  SEQNO _Max;
71  SIZE_T _GapFilled;
72 
73 public:
74  SeqNoDeJitter(SEQNO init = 0){ Reset(init); }
75  void Reset(SEQNO init)
76  { _GapFilled = 0;
77  _Min = _Max = init;
78 #if defined(PLATFORM_DEBUG_BUILD)
79  _Dedup.clear();
80 #endif
81  }
89  bool Insert(SEQNO s)
90  { ASSERT(s > _Min && s != _Max);
91  ASSERT(_Dedup.find(s) == _Dedup.end());
92  if(s > _Max)
93  { _GapFilled += s - _Max - 1;
94  if(_GapFilled == 0)
95  { _Min = _Max = s;
96  ASSERT(_Dedup.size() == 0);
97  return true;
98  }
99 #if defined(PLATFORM_DEBUG_BUILD)
100  _Dedup.insert(_Max);
101 #endif
102  _Max = s;
103  }
104  else
105  { ASSERT(_GapFilled);
106  _GapFilled--;
107  if(_GapFilled == 0)
108  { _Min = _Max;
109 #if defined(PLATFORM_DEBUG_BUILD)
110  _Dedup.clear();
111 #endif
112  return true;
113  }
114 #if defined(PLATFORM_DEBUG_BUILD)
115  _Dedup.insert(s);
116 #endif
117  }
118  return false;
119  }
120  SEQNO DeJittered() const { return _Min; }
121  bool IsJittering() const { return _Min != _Max; }
122  SIZE_T GetMissingCount() const { ASSERT(_Max - _Min >= _GapFilled); return _Max - _Min - _GapFilled; }
123  auto SeqNoMax() const { return _Min; }
124  auto SeqNoMin() const { return _Max; }

308  {
309  UINT expected_seq = _SeqInc(_LastReadSeq);
310  if(_JitterBuf.GetSize())
311  { // pop from jitter buffer if available
312  UINT idx = _SeqMinus(expected_seq, _JitterBufBaseSeq);
313  if(idx < _JitterBuf.GetSize() && _JitterBuf[idx].IsNotNull())
314  {
315  _LastReadSeq = expected_seq;
316  *t = std::move(_JitterBuf[idx].Obj);
317  // shrink jitter buffer
318  if(idx + 1 == _JitterBuf.GetSize()){ _JitterBuf.ShrinkSize(0); }
319  else if(_JitterBuf.GetSize() > JITTER_BUFSIZE_MAX/2 && idx > _JitterBuf.GetSize()/2)
320  { _JitterBuf.erase(0, idx+1);
321  _JitterBufBaseSeq = (idx + 1 + _JitterBufBaseSeq)&SEQ_BITMASK;
322  }
323  return true;
324  }
325  }
326 
327  SeqT pop;
328  while(_Queue.Pop(pop, args...))
329  {
330  if(pop.Seq() == expected_seq) // no jitter
331  {
332  _LastReadSeq = expected_seq;
333  *t = std::move(pop.Obj);
334  return true;
335  }
336  // jittered
337  ASSERT(pop.IsNotNull());

339  if(_JitterBuf.GetSize())
